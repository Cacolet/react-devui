# zh-Hant

最简单的用法。

```tsx
import { useState } from 'react';

import { DAutoComplete, DInput, DTextarea } from '@react-devui/ui';

export default function Demo() {
  const [value1, setValue1] = useState('');
  const [options1, setOptions1] = useState([]);

  const [value2, setValue2] = useState('');
  const [options2, setOptions2] = useState([]);

  return (
    <>
      <DAutoComplete
        dOptions={options1}
        onSelectChange={(val) => {
          setValue1(val);
        }}
      >
        <DInput
          dModel={value1}
          onModelChange={(val) => {
            setValue1(val);
            setOptions1(
              Array(val ? 3 : 0)
                .fill(0)
                .map((n, i) => ({
                  value: Array(i + 1)
                    .fill(val)
                    .join(''),
                }))
            );
          }}
        />
      </DAutoComplete>
      <DAutoComplete
        dOptions={options2}
        onSelectChange={(val) => {
          setValue2(val);
        }}
      >
        <DTextarea
          dModel={value2}
          rows="5"
          onModelChange={(val) => {
            setValue2(val);
            setOptions2(
              Array(val ? 3 : 0)
                .fill(0)
                .map((n, i) => ({
                  value: Array(i + 1)
                    .fill(val)
                    .join(''),
                }))
            );
          }}
        />
      </DAutoComplete>
    </>
  );
}
```
