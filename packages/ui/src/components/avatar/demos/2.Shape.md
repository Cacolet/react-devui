# en-US

Set the shape via `dShape`.

# zh-Hant

通过 `dShape` 设置形状。

```tsx
import { DAvatar } from '@react-devui/ui';

export default function Demo() {
  return (
    <>
      <DAvatar dText="U" dShape="square"></DAvatar>
      <DAvatar dText="U" dShape={8}></DAvatar>
    </>
  );
}
```
