# en-US

Example of a card using an image to reinforce the content.

# zh-Hant

该例使用图像来增强内容的卡片。

```tsx
import { DCard, DCardContent, DCardActions } from '@react-devui/ui';
import { EditOutlined, DeleteOutlined, EllipsisOutlined } from '@react-devui/ui/icons';

export default function Demo() {
  return (
    <DCard style={{ width: 300 }}>
      <div style={{ margin: '-1px -1px 0 -1px' }}>
        <img style={{ width: '100%', borderRadius: 'var(--d-border-radius) var(--d-border-radius) 0 0' }} src="/assets/bg.png" alt="bg" />
      </div>
      <DCardContent>
        <div className="app-demo-container">
          <div>Some contents...</div>
          <div>Some contents...</div>
          <div>Some contents...</div>
        </div>
      </DCardContent>
      <DCardActions
        dActions={[
          <button title="edit">
            <EditOutlined />
          </button>,
          <button title="delete">
            <DeleteOutlined />
          </button>,
          <button title="more">
            <EllipsisOutlined />
          </button>,
        ]}
      ></DCardActions>
    </DCard>
  );
}
```
