# en-US

This example shows the effect of different style configurations.

# zh-Hant

该示例展示了不同样式配置下的效果。

```tsx
import { useState } from 'react';

import { DCard, DCardContent, DRadioGroup } from '@react-devui/ui';
import { EditOutlined, DeleteOutlined, EllipsisOutlined } from '@react-devui/ui/icons';

export default function Demo() {
  const [border, setBorder] = useState(true);
  const [shadow, setShadow] = useState(false);

  return (
    <>
      <DRadioGroup
        dOptions={[true, false].map((border) => ({
          label: border ? 'Border' : 'No border',
          value: border,
        }))}
        dModel={border}
        dType="outline"
        onModelChange={setBorder}
      />
      <br />
      <DRadioGroup
        dOptions={[true, false, 'hover'].map((shadow) => ({
          label: shadow === true ? 'Shadow' : shadow === false ? 'No shadow' : 'Hover shadow',
          value: shadow,
        }))}
        dModel={shadow}
        dType="outline"
        onModelChange={setShadow}
      />
      <br />
      <DCard
        style={{ width: 300 }}
        dBorder={border}
        dShadow={shadow}
        dActions={[
          <button title="edit">
            <EditOutlined />
          </button>,
          <button title="delete">
            <DeleteOutlined />
          </button>,
          <button title="more">
            <EllipsisOutlined />
          </button>,
        ]}
      >
        <DCardContent>
          <div className="app-demo-container">
            <div>Some contents...</div>
            <div>Some contents...</div>
            <div>Some contents...</div>
          </div>
        </DCardContent>
      </DCard>
    </>
  );
}
```
