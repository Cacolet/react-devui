# en-US

Nested lists are supported.

# zh-Hant

支持嵌套列表。

```tsx
import { DVirtualScroll } from '@react-devui/ui';

const list = [
  {
    label: 'Group 1',
    children: Array(20)
      .fill(0)
      .map((item, index) => ({
        label: `Item1 ${index}`,
      })),
  },
  {
    label: 'Group 2',
    children: [],
  },
  {
    label: 'Group 3',
    children: Array(100000)
      .fill(0)
      .map((item, index) => ({
        label: `Item3 ${index}`,
      })),
  },
];
export default function Demo() {
  return (
    <DVirtualScroll
      style={{ height: 192, overflow: 'auto' }}
      dList={list}
      dItemRender={(item, index, { iARIA, iChildren }) => {
        if (item.children) {
          return (
            <ul key={item.label} role="group">
              <li key={item.label} style={{ position: 'relative', left: -32, height: 32 }} role="presentation">
                {item.label}
              </li>
              {item.children.length === 0 ? <li style={{ height: 32 }}>No Data</li> : iChildren}
            </ul>
          );
        }

        return (
          <li {...iARIA} key={item.label} style={{ height: 32 }}>
            {item.label}
          </li>
        );
      }}
      dItemSize={(item) => {
        if (item.children && item.children.length === 0) {
          return 64;
        }
        return 32;
      }}
      dItemNested={(item) => item.children}
      dItemKey={(item) => item.label}
      dSize={192}
    ></DVirtualScroll>
  );
}
```
